%% Pandoc LaTeX template for IOS Press Book Articles
%% Based on the IOS-Book-Article class (https://github.com/vtex-soft/texsupport.IOS-Book-Article)
%%
%% Required YAML front-matter keys:
%%   title          : Full paper title
%%   runningtitle   : Short title for running header
%%   runningauthor  : Author(s) for running header, e.g. "Smith et al."
%%   authors        : List of author objects (see below)
%%   addresses      : List of affiliation objects (see below)
%%   abstract       : Paper abstract (block scalar)
%%   keywords       : List of keyword strings
%%
%% Optional front-matter keys:
%%   booktitle      : Title of the book this chapter belongs to
%%   bookeditors    : Editors of the book
%%   bibliography   : Path to a .bib file (relative to the paper's directory)
%%
%% Author object fields:
%%   fnms           : First / middle name(s)
%%   snm            : Surname
%%   label          : Affiliation label, e.g. "A"  (optional for single author)
%%   orcid          : ORCID iD, e.g. "0000-0000-0000-0000" (optional)
%%   corresponding  : true/false – adds a corresponding-author note (optional)
%%   note           : Free-form footnote text for the author (optional)
%%
%% Address object fields:
%%   label          : Matches the label(s) used in authors (optional for single affil.)
%%   institution    : Affiliation text

\documentclass{IOS-Book-Article}

\usepackage{mathptmx}
\usepackage{graphicx}
\usepackage{amsmath}
\usepackage{amssymb}
\usepackage{longtable}
\usepackage{booktabs}
\usepackage{array}
\usepackage{calc}
\usepackage{listings}

%% Code highlighting (from pandoc)
$if(highlighting-macros)$
$highlighting-macros$
$endif$

%%%%%%%%%%% Put your definitions here

%% Pandoc tight-list fix
\providecommand{\tightlist}{%
  \setlength{\itemsep}{0pt}\setlength{\parskip}{0pt}}

%%%%%%%%%%% End of definitions

$if(booktitle)$
\booktitle{$booktitle$}
$endif$
$if(bookeditors)$
\bookeditors{$bookeditors$}
$endif$

\begin{document}

\pagestyle{headings}

\begin{frontmatter}

$if(title)$
\title{$title$}
$endif$
$if(runningtitle)$
\runningtitle{$runningtitle$}
$endif$

$-- Multiple structured authors
$if(authors)$
$for(authors)$
\author$if(authors.label)$[$authors.label$]$endif${\fnms{$authors.fnms$} \snm{$authors.snm$}$if(authors.orcid)$\orcid{$authors.orcid$}$endif$$if(authors.corresponding)$\thanks{Corresponding Author: $authors.fnms$ $authors.snm$$if(authors.note)$, $authors.note$$endif$.}$endif$}$sep$,
$endfor$
$else$
$-- Fallback: simple author string(s)
$if(author)$
\author{$for(author)$$author$$sep$ \and $endfor$}
$endif$
$endif$

$if(runningauthor)$
\runningauthor{$runningauthor$}
$endif$

$-- Affiliations
$if(addresses)$
$for(addresses)$
\address$if(addresses.label)$[$addresses.label$]$endif${$addresses.institution$}
$endfor$
$else$
$-- Fallback: single address string
$if(address)$
\address{$address$}
$endif$
$endif$

$if(abstract)$
\begin{abstract}
$abstract$
\end{abstract}
$endif$

$if(keywords)$
\begin{keyword}
$for(keywords)$$keywords$$sep$\sep $endfor$
\end{keyword}
$endif$

\end{frontmatter}

$body$

$-- Bibliography
$if(bibliography)$
\bibliographystyle{vancouver}
\bibliography{$bibliography$}
$endif$

\end{document}
